if not os.path.exists(output_folder):
    os.makedirs(output_folder)

# Obtener la lista de todas las imágenes
imagenes = os.listdir(carpeta_imagenes)

# Dividir las imágenes en subcarpetas de 500 imágenes cada una
num_imagenes_por_carpeta = 500
for i in range(0, len(imagenes), num_imagenes_por_carpeta):
    subcarpeta = os.path.join(output_folder, f'carpeta_{i // num_imagenes_por_carpeta + 1}')
    os.makedirs(subcarpeta, exist_ok=True)
    for imagen in imagenes[i:i + num_imagenes_por_carpeta]:
        shutil.move(os.path.join(carpeta_imagenes, imagen), subcarpeta)

print('Imágenes divididas en subcarpetas correctamente.')
